#include <iostream>
using namespace std;
int main() {
  int prev = -1, cur;
  while (cin >> cur) {
    if (cur == prev) {
      continue;
    }
    prev = cur;
    cout << cur << " ";
  }
  cout << "-1" << endl;
  return 0;
}

#include <iostream>
using namespace std;
int main() {
  float a1, b1, c1, a2, b2, c2;
  cout << "Nhập hệ số a1, b1, c1 của phương trình thứ nhất: ";
  cin >> a1 >> b1 >> c1;
  cout << "Nhập hệ số a2, b2, c2 của phương trình thứ hai: ";
  cin >> a2 >> b2 >> c2;
  float D = a1 * b2 - a2 * b1;
  float Dx = c1 * b2 - c2 * b1;
  float Dy = a1 * c2 - a2 * c1;
  if (D == 0) {
    if (Dx == 0 && Dy == 0) {
      cout << "Hệ phương trình có vô số nghiệm." << endl;
    } else {
      // Hệ phương trình vô nghiệm
      cout << "Hệ phương trình vô nghiệm." << endl;
    }
  } else {
    // Hệ phương trình có một nghiệm duy nhất
    float x = Dx / D;
    float y = Dy / D;
    cout << "Hệ phương trình có một nghiệm duy nhất: " << endl;
    cout << "x = " << x << endl;
    cout << "y = " << y << endl;
  }
  return 0;
}

#include <iostream>
using namespace std;
int main() {
  int n;
  do {
    cout << "Nhập vào một số trong khoảng 0..9: ";
    cin >> n;
    if (n >= 0 && n <= 9) {
      switch (n) {
        case 0:
          cout << "Vừa nhập chữ số không" << endl;
          break;

          cout << "Vừa nhập chữ số một" << endl;
